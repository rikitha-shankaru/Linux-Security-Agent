# Commands to run in your Linux VM (copy-paste these)

# 1. Navigate to project directory
cd ~/linux_security_agent

# 2. Create virtual environment
python3 -m venv venv

# 3. Activate virtual environment
source venv/bin/activate

# 4. Install Python dependencies
pip install --upgrade pip
pip install numpy pandas scikit-learn psutil rich click requests

# 5. Install BCC tools for eBPF (optional but recommended)
sudo apt-get update
sudo apt-get install -y bpfcc-tools python3-bpfcc linux-headers-$(uname -r)

# 6. Train models with REAL syscall data
sudo python3 core/enhanced_security_agent.py --train-models

# 7. After training completes, verify models
ls -lh ~/.cache/security_agent/

# 8. Run the agent with dashboard
sudo python3 core/enhanced_security_agent.py --dashboard
